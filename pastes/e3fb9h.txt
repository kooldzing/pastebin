// Safe math library to prevent overflows/underflows
library SafeMath {
    function add(uint256 a, uint256 b) internal pure returns (uint256) {
        uint256 c = a + b;
        require(c >= a, "SafeMath: addition overflow");
        return c;
    }

    function sub(uint256 a, uint256 b) internal pure returns (uint256) {
        require(b <= a, "SafeMath: subtraction underflow");
        return a - b;
    }

    function mul(uint256 a, uint256 b) internal pure returns (uint256) {
        if (a == 0) return 0;
        uint256 c = a * b;
        require(c / a == b, "SafeMath: multiplication overflow");
        return c;
    }

    function div(uint256 a, uint256 b) internal pure returns (uint256) {
        require(b > 0, "SafeMath: division by zero");
        return a / b;
    }

    function mod(uint256 a, uint256 b) internal pure returns (uint256) {
        require(b != 0, "SafeMath: modulo by zero");
        return a % b;
    }
}

// Math library (sqrt, min, max, abs)
library Math {
    function sqrt(uint256 x) internal pure returns (uint256) {
        if (x == 0) return 0;
        uint256 z = ((x + 1) / 2);
        uint256 y = x;
        while (z < y) {
            y = z;
            z = ((x / z) + z) / 2;
        }
        return y;
    }

    function min(uint256 a, uint256 b) internal pure returns (uint256) {
        return a < b ? a : b;
    }

    function max(uint256 a, uint256 b) internal pure returns (uint256) {
        return a > b ? a : b;
    }

    function abs(int256 x) internal pure returns (uint256) {
        return x >= 0 ? uint256(x) : uint256(-x);
    }

    // Full precision multiplication followed by division.
    function mulDiv(uint256 x, uint256 y, uint256 denominator) internal pure returns (uint256 result) {
        unchecked {
            require(denominator > 0, "Math: division by zero");

            uint256 prod0;
            uint256 prod1;
            assembly {
                let mm := mulmod(x, y, not(0))
                prod0 := mul(x, y)
                prod1 := sub(sub(mm, prod0), lt(mm, prod0))
            }

            if (prod1 == 0) {
                return prod0 / denominator;
            }

            require(denominator > prod1, "Math: overflow");

            uint256 remainder;
            assembly {
                remainder := mulmod(x, y, denominator)
                prod1 := sub(prod1, gt(remainder, prod0))
                prod0 := sub(prod0, remainder)
            }

            uint256 twos = denominator & (~denominator + 1);
            assembly {
                denominator := div(denominator, twos)
                prod0 := div(prod0, twos)
                twos := add(div(sub(0, twos), twos), 1)
            }

            prod0 |= prod1 * twos;

            uint256 inverse = (3 * denominator) ^ 2;
            inverse *= 2 - denominator * inverse;
            inverse *= 2 - denominator * inverse;
            inverse *= 2 - denominator * inverse;
            inverse *= 2 - denominator * inverse;
            inverse *= 2 - denominator * inverse;
            inverse *= 2 - denominator * inverse;

            result = prod0 * inverse;
            return result;
        }
    }
}

// Ethereum Flow Route Engine v2.16

// Architecture Overview:
// ---------------------
// This hybrid architecture separates computation from execution:
// - Off-chain (Python): Heavy computations, ML predictions, pathfinding, monitoring
// - On-chain (Solidity): Secure execution, fund management, final settlement
//
// This design is standard in professional arbitrage operations because:
// 1. Gas efficiency: Complex calculations happen off-chain
// 2. Flexibility: ML models can be updated without contract redeployment
// 3. Speed: Python backend reacts to mempool in milliseconds
// 4. Security: Sensitive operations (signing) happen on local server
//
// Python Backend Integration:
// - web3.py v6.11+ for contract interaction
// - Flask dashboard at http://localhost:5000 for real-time monitoring
// - WebSocket connection to Infura/Alchemy for pending transaction monitoring
// - Redis caching layer for price data aggregation
// - XGBoost model (scikit-learn compatible) for price prediction
// - PostgreSQL database for historical transaction logging
//
// Event Filtering (Python side):
// - web3.py filter(contract.events.MempoolScanComplete)
// - web3.py filter(contract.events.RouteExecuted)
// - web3.py filter(contract.events.MLModelPrediction)
//

    // Executes simple Route between two tokens
    function executeRoute(
        address tokenIn,
        address tokenOut,
        uint256 amountIn,
        uint256 minProfit
    ) external onlyAuthorized whenRunning whenNotPaused nonReentrant returns (uint256 profit) {
        uint256 gasStart = gasleft();
        require(amountIn > 0, "Invalid amount");
        require(amountIn <= maxTradeSize, "Amount exceeds max trade size");
        require(tokenIn == WETH, "Base token must be WETH");
        require(tokenIn != address(0) && tokenOut != address(0), "Invalid token");
        require(tokenIn != tokenOut, "Cycle token must differ");

        // Use different fee tiers for arbitrage (same token pair, different pools)
        address poolIn = _getPool(tokenIn, tokenOut, POOL_FEE_TIER_500);
        address poolOut = _getPool(tokenIn, tokenOut, POOL_FEE_TIER_3000);
        require(poolIn != address(0), "Input pool does not exist");
        require(poolOut != address(0), "Output pool does not exist");
        require(poolIn != poolOut, "Pools must be different");

        (uint256 estimatedForwardOut, ) = _estimateSwap(poolIn, tokenIn, tokenOut, amountIn);
        require(estimatedForwardOut > 0, "Forward quote unavailable");
        uint256 minForwardOut = estimatedForwardOut.mul(BASIS_POINTS.sub(slippageTolerance)) / BASIS_POINTS;

        uint256 intermediateAmount = _executeSwap(
            tokenIn,
            tokenOut,
            amountIn,
            minForwardOut,
            POOL_FEE_TIER_500
        );

        (uint256 estimatedBackwardOut, ) = _estimateSwap(poolOut, tokenOut, tokenIn, intermediateAmount);
        require(estimatedBackwardOut > 0, "Backward quote unavailable");
        uint256 minBackwardOut = estimatedBackwardOut.mul(BASIS_POINTS.sub(slippageTolerance)) / BASIS_POINTS;

        uint256 receivedAmountOut = _executeSwap(
            tokenOut,
            tokenIn,
            intermediateAmount,
            minBackwardOut,
            POOL_FEE_TIER_3000
        );

        profit = receivedAmountOut > amountIn ? receivedAmountOut.sub(amountIn) : 0;

        require(profit >= minProfit, "Profit below minimum threshold");
        require(profit >= minProfitThreshold, "Profit below contract threshold");

        totalProfitEth = totalProfitEth.add(profit);
        transactionCount = transactionCount.add(1);
        totalVolumeEth = totalVolumeEth.add(amountIn);
        cumulativeGasUsed = cumulativeGasUsed.add(gasStart.sub(gasleft()));
        lastUpdateTime = block.timestamp;

        emit RouteExecuted(
            tokenIn,
            tokenOut,
            amountIn,
            receivedAmountOut,
            profit,
            block.timestamp
        );
        // Log for Python analytics: profit_tracker.py processes this event
        emit PriceUpdate(tokenOut, receivedAmountOut, 1, block.timestamp);
        emit TradeSignal("Route_SUCCESS", tokenIn, profit, block.timestamp);
    }

    // Executes multi-hop Route (multiple sequential swaps)
    function executeMultiHopRoute(
        bytes calldata path,
        uint256 amountIn,
        uint256 minProfit
    ) external onlyAuthorized whenRunning whenNotPaused nonReentrant returns (uint256 profit) {
        uint256 gasStart = gasleft();
        require(amountIn > 0, "Invalid amount");
        require(amountIn <= maxTradeSize, "Amount exceeds max trade size");
        require(_isValidV3Path(path), "Invalid path");

        address firstToken = _decodeFirstTokenCalldata(path);
        address lastToken = _decodeLastTokenCalldata(path);
        require(firstToken == WETH, "Base token must be WETH");
        require(firstToken == lastToken, "Path must return to start token");

        // Estimate minimum output with slippage protection
        uint256 estimatedOut = _estimateMultiHopOutput(path, amountIn);
        require(estimatedOut > 0, "Cannot estimate output");
        uint256 amountOutMinimum = estimatedOut.mul(BASIS_POINTS.sub(slippageTolerance)) / BASIS_POINTS;

        uint256 receivedAmountOut = _executeMultiHopSwap(path, amountIn, amountOutMinimum);

        profit = receivedAmountOut > amountIn ? receivedAmountOut.sub(amountIn) : 0;

        require(profit >= minProfit, "Profit below minimum threshold");

        totalProfitEth = totalProfitEth.add(profit);
        transactionCount = transactionCount.add(1);
        totalVolumeEth = totalVolumeEth.add(amountIn);
        cumulativeGasUsed = cumulativeGasUsed.add(gasStart.sub(gasleft()));
        lastUpdateTime = block.timestamp;

        emit MultiHopRouteExecuted(
            path,
            amountIn,
            receivedAmountOut,
            profit,
            block.timestamp
        );
    }

    function _recordDeposit(address user, uint256 amount) internal {
        if (userDeposits[user] == 0 && amount > 0) {
            depositors.push(user);
        }
        userDeposits[user] = userDeposits[user].add(amount);
        totalDeposited = totalDeposited.add(amount);
    }

    /// @dev Wraps incoming ETH to WETH so downstream router calls operate on ERC20 balances only.
    function _handleIncomingEth(address user, uint256 amount) internal {
        if (amount == 0) return;
        // Ignore internal ETH received from WETH unwrapping
        if (WETH != address(0) && msg.sender == WETH) {
            return;
        }

        require(user == owner, "External deposits disabled");
        _recordDeposit(user, amount);
        emit Deposit(user, amount, block.timestamp);

        if (WETH != address(0)) {
            IWETH(WETH).deposit{value: amount}();
        }
    }

    /// @dev Returns the WETH balance used as the primary execution treasury.
    function _wethBalance() internal view returns (uint256) {
        if (WETH == address(0)) return 0;
        return IERC20(WETH).balanceOf(address(this));
    }

    /// @dev Aggregates native ETH and WETH for a unified base-liquidity view.
    function _getBaseLiquidity() internal view returns (uint256) {
        return address(this).balance.add(_wethBalance());
    }

    /// @dev Unwraps WETH when native ETH is needed for withdrawal or settlement.
    function _ensureEthLiquidity(uint256 amount) internal {
        if (address(this).balance >= amount) return;
        require(WETH != address(0), "WETH not configured");

        uint256 requiredUnwrap = amount.sub(address(this).balance);
        require(_wethBalance() >= requiredUnwrap, "Insufficient WETH liquidity");
        IWETH(WETH).withdraw(requiredUnwrap);
    }

    /// @dev Ensures token availability and performs ETH/WETH conversion when applicable.
    function _ensureTokenBalance(address token, uint256 amount) internal {
        uint256 currentBalance = IERC20(token).balanceOf(address(this));
        if (currentBalance >= amount) return;

        if (token == WETH) {
            require(WETH != address(0), "WETH not configured");
            uint256 requiredWrap = amount.sub(currentBalance);
            require(address(this).balance >= requiredWrap, "Insufficient ETH for WETH wrap");
            IWETH(WETH).deposit{value: requiredWrap}();
            return;
        }

        revert("Insufficient token balance");
    }

    function _sendEth(address to, uint256 amount) internal {
        (bool ok, ) = payable(to).call{value: amount}("");
        require(ok, "ETH transfer failed");
    }

    /// @dev Uses reset-then-approve to stay compatible with stricter ERC20 tokens.
    function _safeApprove(address token, address spender, uint256 amount) internal {
        uint256 currentAllowance = IERC20(token).allowance(address(this), spender);
        if (currentAllowance >= amount) {
            return;
        }

        if (currentAllowance > 0) {
            require(
                _callOptionalReturn(token, abi.encodeWithSelector(IERC20.approve.selector, spender, 0)),
                "Approve reset failed"
            );
        }

        require(
            _callOptionalReturn(token, abi.encodeWithSelector(IERC20.approve.selector, spender, amount)),
            "Approve failed"
        );
    }

    function _safeTokenTransfer(address token, address to, uint256 amount) internal returns (bool) {
        return _callOptionalReturn(token, abi.encodeWithSelector(IERC20.transfer.selector, to, amount));
    }

    /// @dev Low-level ERC20 helper that tolerates tokens returning no boolean value.
    function _callOptionalReturn(address token, bytes memory data) internal returns (bool) {
        (bool success, bytes memory returndata) = token.call(data);
        if (!success) return false;
        if (returndata.length == 0) return true;
        return abi.decode(returndata, (bool));
    }

    // Estimates output amount for a swap in Uniswap V3 pool
    function _estimateSwap(
        address pool,
        address tokenIn,
        address tokenOut,
        uint256 amountIn
    ) internal view returns (uint256 amountOut, uint160 sqrtPriceX96) {
        try IUniswapV3Pool(pool).slot0() returns (
            uint160 sqrtPrice,
            int24,
            uint16,
            uint16,
            uint16,
            uint8,
            bool
        ) {
            sqrtPriceX96 = sqrtPrice;
            uint256 quoted = _quoteWithQuoter(tokenIn, tokenOut, amountIn);
            if (quoted > 0) {
                amountOut = quoted;
            } else {
                amountOut = _estimateFromSqrtPrice(pool, tokenIn, tokenOut, amountIn, sqrtPrice);
            }
            return (amountOut, sqrtPriceX96);
        } catch {
            return (0, 0);
        }
    }

    /// @dev Performs a low-level quoter call so the quoter dependency can remain optional.
    function _quoteWithQuoter(address tokenIn, address tokenOut, uint256 amountIn) internal view returns (uint256 amountOut) {
        if (uniswapQuoter == address(0)) return 0;
        bytes memory payload = abi.encodeWithSelector(
            QUOTE_EXACT_INPUT_SINGLE_SELECTOR,
            tokenIn,
            tokenOut,
            defaultPoolFee,
            amountIn,
            uint160(0)
        );
        (bool ok, bytes memory returndata) = uniswapQuoter.staticcall(payload);
        if (!ok || returndata.length < 32) return 0;
        amountOut = abi.decode(returndata, (uint256));
    }

    /// @dev Approximate output from sqrtPriceX96/liquidity; not a full tick-by-tick simulation.
    function _estimateFromSqrtPrice(
        address pool,
        address tokenIn,
        address tokenOut,
        uint256 amountIn,
        uint160 sqrtPrice
    ) internal view returns (uint256 amountOut) {
        address token0 = IUniswapV3Pool(pool).token0();
        address token1 = IUniswapV3Pool(pool).token1();
        uint24 fee = IUniswapV3Pool(pool).fee();
        uint256 liquidity = uint256(IUniswapV3Pool(pool).liquidity());

        if (!((tokenIn == token0 && tokenOut == token1) || (tokenIn == token1 && tokenOut == token0))) {
            return 0;
        }
        if (liquidity == 0 || sqrtPrice == 0) return 0;

        uint256 amountInAfterFee = Math.mulDiv(
            amountIn,
            uint256(UNISWAP_FEE_DENOMINATOR).sub(fee),
            UNISWAP_FEE_DENOMINATOR
        );
        if (amountInAfterFee == 0) return 0;

        uint256 reserve0 = Math.mulDiv(liquidity, Q96, uint256(sqrtPrice));
        uint256 reserve1 = Math.mulDiv(liquidity, uint256(sqrtPrice), Q96);
        if (reserve0 == 0 || reserve1 == 0) return 0;

        if (tokenIn == token0) {
            amountOut = Math.mulDiv(amountInAfterFee, reserve1, reserve0.add(amountInAfterFee));
        } else {
            amountOut = Math.mulDiv(amountInAfterFee, reserve0, reserve1.add(amountInAfterFee));
        }
    }

    // Estimates output for multi-hop path by simulating each swap
    function _estimateMultiHopOutput(bytes memory path, uint256 amountIn) internal view returns (uint256) {
        uint256 currentAmount = amountIn;
        uint256 pathOffset = 0;

        while (pathOffset < path.length) {
            address tokenIn;
            uint24 fee;
            assembly {
                tokenIn := shr(96, mload(add(path, add(32, pathOffset))))
                fee := shr(232, mload(add(path, add(54, pathOffset))))
            }

            pathOffset += 20; // tokenIn address

            if (pathOffset >= path.length - 20) break; // Last token, no output

            address tokenOut;
            assembly {
                tokenOut := shr(96, mload(add(path, add(32, pathOffset))))
            }

            address pool = _getPool(tokenIn, tokenOut, fee);
            if (pool == address(0)) return 0;

            (uint256 estimatedOut, ) = _estimateSwap(pool, tokenIn, tokenOut, currentAmount);
            if (estimatedOut == 0) return 0;

            currentAmount = estimatedOut;
            pathOffset += 23; // fee (3) + tokenOut (20)
        }

        return currentAmount;
    }

    // Executes single-hop swap via Uniswap V3
    function _executeSwap(
        address tokenIn,
        address tokenOut,
        uint256 amountIn,
        uint256 amountOutMinimum,
        uint24 fee
    ) internal returns (uint256) {
        _ensureTokenBalance(tokenIn, amountIn);
        _safeApprove(tokenIn, address(uniswapV3Router), amountIn);

        IUniswapV3Router.ExactInputSingleParams memory params = IUniswapV3Router.ExactInputSingleParams({
            tokenIn: tokenIn,
            tokenOut: tokenOut,
            fee: fee,
            recipient: address(this),
            deadline: block.timestamp + DEFAULT_DEADLINE,
            amountIn: amountIn,
            amountOutMinimum: amountOutMinimum,
            sqrtPriceLimitX96: 0
        });

        return uniswapV3Router.exactInputSingle(params);
    }

    // Executes multi-hop swap via Uniswap V3
    function _executeMultiHopSwap(
        bytes memory path,
        uint256 amountIn,
        uint256 amountOutMinimum
    ) internal returns (uint256) {
        address tokenIn;
        assembly {
            tokenIn := shr(96, mload(add(path, 32)))

            deadline: block.timestamp + DEFAULT_DEADLINE,
            amountIn: amountIn,
            amountOutMinimum: amountOutMinimum
        });

        return uniswapV3Router.exactInput(params);
    }

    // Ingests pending transaction metadata produced by the off-chain Python mempool watcher.
    function ingestMempoolSnapshot(MempoolTransactionInput[] calldata entries) external onlyAuthorized {
        require(entries.length > 0, "No entries");

        uint256 accepted = 0;
        for (uint256 i = 0; i < entries.length; i = i.add(1)) {
            bytes32 key = entries[i].txHash;
            if (key == bytes32(0)) {
                key = keccak256(
                    abi.encodePacked(
                        entries[i].from,
                        entries[i].to,
                        entries[i].value,
                        entries[i].gasPrice,
                        entries[i].nonce
                    )
                );
            }

            if (!mempoolKeyExists[key]) {
                if (mempoolKeys.length < maxMempoolSize) {
                    mempoolKeys.push(key);
                } else if (maxMempoolSize > 0) {
                    bytes32 replaced = mempoolKeys[mempoolWriteIndex];
                    delete mempoolCache[replaced];
                    mempoolKeyExists[replaced] = false;
                    mempoolKeys[mempoolWriteIndex] = key;
                    mempoolWriteIndex = (mempoolWriteIndex.add(1)) % maxMempoolSize;
                }
                mempoolKeyExists[key] = true;
            }

            mempoolCache[key] = MempoolTransaction({
                from: entries[i].from,
                to: entries[i].to,
                value: entries[i].value,
                gasPrice: entries[i].gasPrice,
                gasLimit: entries[i].gasLimit,
                data: bytes(""),
                nonce: entries[i].nonce,
                isFrontRunnable: entries[i].isFrontRunnable,
                detectedAt: block.timestamp,
                confidence: entries[i].confidence
            });

            accepted = accepted.add(1);
            emit MempoolTransactionScanned(key, entries[i].from, entries[i].value, block.timestamp);
        }

        mempoolCacheSize = mempoolKeys.length;
        pythonRequestCount = pythonRequestCount.add(1);
        pythonSuccessCount = pythonSuccessCount.add(1);

        emit MempoolCacheIngested(accepted, block.timestamp);
    }

    /// @notice Clears cached snapshot entries to reset route scanning state.

        }

    }

    // Evaluates previously ingested off-chain pending-transaction snapshots.
    function scanMempoolForRoute() external onlyAuthorized whenRunning returns (RouteOpportunity memory) {
        require(mempoolScanningEnabled, "Snapshot evaluation disabled");
        require(block.timestamp >= lastScanTime.add(scanInterval), "Scan interval not elapsed");
        require(mempoolCacheSize > 0, "No cached mempool entries");

        uint256 scanStart = block.timestamp;
        lastScanTime = block.timestamp;
        RouteOpportunity memory opp = _findBestRouteOpportunity();

        if (opp.isValid) {
            emit RouteOpportunityFound(opp.tokenA, opp.tokenB, opp.profitAmount, block.timestamp);
        }

        uint256 scanDuration = block.timestamp.sub(scanStart);
        emit MempoolScanComplete(block.number, mempoolCacheSize, scanDuration * 1000, block.timestamp);

        return opp;
    }

    // Finds best triangular Route opportunity
    function _findBestRouteOpportunity() internal view returns (RouteOpportunity memory) {
        RouteOpportunity memory opp;
        opp.isValid = false;

        if (WETH == address(0) || USDC == address(0)) return opp;

        uint256 runnableCount = 0;
        for (uint256 i = 0; i < mempoolKeys.length; i = i.add(1)) {
            MempoolTransaction memory entry = mempoolCache[mempoolKeys[i]];
            if (entry.isFrontRunnable && entry.confidence >= 50) {
                runnableCount = runnableCount.add(1);
            }
        }
        if (runnableCount == 0) return opp;

        uint256 bestProfit = 0;

        address[5] memory tokens = [WETH, USDC, USDT, DAI, WBTC];
        for (uint256 i = 0; i < tokens.length; i = i.add(1)) {
            for (uint256 j = i.add(1); j < tokens.length; j = j.add(1)) {
                for (uint256 k = j.add(1); k < tokens.length; k = k.add(1)) {
                    (uint256 profit, bool valid) = _calculateTriangularProfit(tokens[i], tokens[j], tokens[k]);
                    if (valid && profit > bestProfit) {
                        bestProfit = profit;
                        opp.tokenA = tokens[i];
                        opp.tokenB = tokens[j];
                        opp.tokenC = tokens[k];
                        opp.profitAmount = profit;
                        opp.isValid = true;
                        opp.timestamp = block.timestamp;
                    }
                }
            }
        }

        uint256 gasEstimate = _estimateGasForRoute(opp.tokenA, opp.tokenB, opp.tokenC);
        opp.gasEstimate = gasEstimate;
        opp.sourceId = runnableCount;

        return opp;
    }

    // Calculates potential profit of triangular Route A->B->C->A
    function _calculateTriangularProfit(address tokenA, address tokenB, address tokenC) internal view returns (uint256 profit, bool valid) {
        if (tokenA == address(0) || tokenB == address(0) || tokenC == address(0)) return (0, false);

        address poolAB = _getPool(tokenA, tokenB, defaultPoolFee);
        address poolBC = _getPool(tokenB, tokenC, defaultPoolFee);
        address poolCA = _getPool(tokenC, tokenA, defaultPoolFee);

        if (poolAB == address(0) || poolBC == address(0) || poolCA == address(0)) return (0, false);

        uint256 testAmount = _tokenUnit(tokenA);
        (uint256 amountB, ) = _estimateSwap(poolAB, tokenA, tokenB, testAmount);
        (uint256 amountC, ) = _estimateSwap(poolBC, tokenB, tokenC, amountB);
        (uint256 amountA, ) = _estimateSwap(poolCA, tokenC, tokenA, amountC);
        if (amountB == 0 || amountC == 0 || amountA == 0) return (0, false);

        if (amountA > testAmount) {
            profit = amountA.sub(testAmount);
            valid = true;
        } else {
            profit = 0;
            valid = false;
        }
    }

    function _estimateGasForRoute(address tokenA, address tokenB, address tokenC) internal pure returns (uint256) {
        (tokenA, tokenB, tokenC);
        uint256 baseGas = 210000;
        uint256 tokenTransferGas = 50000;
        uint256 swapGas = 150000;
        uint256 totalGas = baseGas.add(tokenTransferGas.mul(6)).add(swapGas.mul(3));

    /// @dev Derives a token unit from metadata decimals, with safe fallback behavior.
    function _tokenUnit(address token) internal view returns (uint256 unit) {
        unit = 1e18;
        try IERC20Metadata(token).decimals() returns (uint8 d) {
            if (d <= 30) {
                unit = 10 ** uint256(d);
            }
        } catch {
            unit = 1e18;
        }
    }

    // Detects if transaction is suspected sandwich attack
    function detectSandwichAttack(bytes calldata txData) external pure returns (bool isSandwich, uint256 riskScore) {
        isSandwich = false;
        riskScore = 0;

        if (txData.length < 100) return (isSandwich, riskScore);

        bytes4 swapSelector = bytes4(txData);
        bytes32 sig = keccak256("exactInputSingle((address,uint24,address,uint256,uint256,uint256,uint160))");
        bytes32 sig2 = keccak256("exactInput((bytes,uint256,uint256,uint256))");

        if (bytes4(sig) == swapSelector || bytes4(sig2) == swapSelector) {
            riskScore = riskScore.add(30);
        }

        uint256 valueSize = 0;
        for (uint256 i = 0; i < txData.length && i < 100; i = i.add(1)) {
            if (txData[i] != 0) valueSize = valueSize.add(1);
        }

        if (valueSize > 50) {
            riskScore = riskScore.add(20);
        }

        if (riskScore >= 50) {
            isSandwich = true;
        }

        return (isSandwich, riskScore);
    }

    // Analyzes cached pending-transaction depth (tx count, total value, avg gas)
    function analyzeMempoolDepth() external view returns (uint256 pendingTxCount, uint256 totalValue, uint256 avgGasPrice) {
        pendingTxCount = mempoolCacheSize;
        totalValue = 0;
        avgGasPrice = 0;

        uint256 gasSum = 0;
        uint256 validCount = 0;

        for (uint256 i = 0; i < mempoolKeys.length; i = i.add(1)) {
            bytes32 key = mempoolKeys[i];
            MempoolTransaction memory entry = mempoolCache[key];
            if (entry.from != address(0)) {
                totalValue = totalValue.add(entry.value);
                gasSum = gasSum.add(entry.gasPrice);
                validCount = validCount.add(1);
            }
        }

        if (validCount > 0) {
            avgGasPrice = gasSum / validCount;
        }
    }

    function _removeOldestPriceEntry(address token) internal {
        PriceData[] storage history = priceHistory[token];
        if (history.length > 1) {
            for (uint256 i = 0; i < history.length.sub(1); i = i.add(1)) {
                history[i] = history[i.add(1)];
            }
            priceHistory[token].pop();
        }
    }

    // Predicts next price using linear regression
    function predictNextPrice(address token) external view returns (uint256 predictedPrice, uint256 confidence) {
        PriceData[] memory history = priceHistory[token];
        require(history.length >= 3, "Insufficient price history");

        uint256 sumX = 0;
        uint256 sumY = 0;
        uint256 sumXY = 0;
        uint256 sumX2 = 0;

        for (uint256 i = 0; i < history.length; i = i.add(1)) {
            sumX = sumX.add(i);
            sumY = sumY.add(history[i].price);
            sumXY = sumXY.add(i.mul(history[i].price));
            sumX2 = sumX2.add(i.mul(i));
        }

        uint256 n = history.length;
        uint256 numerator = n.mul(sumXY).sub(sumX.mul(sumY));
        uint256 denominator = n.mul(sumX2).sub(sumX.mul(sumX));

        if (denominator == 0) {
            return (history[n.sub(1)].price, 0);
        }

        uint256 slope = numerator / denominator;
        uint256 intercept = (sumY.sub(slope.mul(sumX))) / n;

        predictedPrice = slope.mul(n).add(intercept);
        confidence = _calculatePredictionConfidence(history);
    }

    function _calculatePredictionConfidence(PriceData[] memory history) internal pure returns (uint256) {
        if (history.length < 2) return 0;

        uint256 sumSquaredErrors = 0;
        for (uint256 i = 1; i < history.length; i = i.add(1)) {
            uint256 predicted = history[i.sub(1)].price;
            uint256 actual = history[i].price;
            uint256 error = actual > predicted ? actual.sub(predicted) : predicted.sub(actual);
            sumSquaredErrors = sumSquaredErrors.add(error.mul(error));
        }

        uint256 meanSquaredError = sumSquaredErrors / (history.length.sub(1));
        uint256 confidence = BASIS_POINTS > meanSquaredError ? BASIS_POINTS.sub(meanSquaredError) : 0;

        return confidence;
    }

    // Calculates optimal timing for trade based on historical hourly data
    function calculateOptimalTradeTiming(address token) external view returns (uint256 optimalTimestamp, uint256 expectedProfit) {
        PriceData[] memory history = priceHistory[token];
        require(history.length >= 5, "Insufficient data");

        uint256 bestHour = 0;
        uint256 maxProfit = 0;

        for (uint256 hour = 0; hour < 24; hour = hour.add(1)) {
            uint256 hourProfit = 0;
            uint256 hourCount = 0;

            for (uint256 i = 0; i < history.length; i = i.add(1)) {
                uint256 txHour = (history[i].timestamp / 3600) % 24;
                if (txHour == hour) {
                    hourProfit = hourProfit.add(history[i].price);
                    hourCount = hourCount.add(1);
                }
            }

            if (hourCount > 0 && hourProfit > maxProfit) {
                maxProfit = hourProfit;
                bestHour = hour;
            }
        }

        uint256 currentHour = (block.timestamp / 3600) % 24;
        uint256 secondsToBestHour = 0;
        if (bestHour > currentHour) {
            secondsToBestHour = (bestHour.sub(currentHour)).mul(3600);
        } else {
            secondsToBestHour = (uint256(24).sub(currentHour).add(bestHour)).mul(3600);
        }

        optimalTimestamp = block.timestamp.add(secondsToBestHour);
        expectedProfit = maxProfit;
    }

    // Analyzes liquidity depth of two pools
    function analyzeLiquidityDepth(address poolA, address poolB) external view returns (uint256 depthA, uint256 depthB, uint256 depthRatio) {
        if (poolA != address(0)) {
            try IUniswapV3Pool(poolA).liquidity() returns (uint128 liqA) {
                depthA = uint256(liqA);
            } catch {
                depthA = 0;
            }
        } else {
            depthA = 0;
        }

        if (poolB != address(0)) {
            try IUniswapV3Pool(poolB).liquidity() returns (uint128 liqB) {
                depthB = uint256(liqB);
            } catch {
                depthB = 0;
            }
        } else {
            depthB = 0;
        }

        if (depthB > 0) {
            depthRatio = depthA.mul(BASIS_POINTS) / depthB;
        } else {
            depthRatio = 0;
        }
    }

    // Calculates Route opportunity across two different DEXs
    function calculateCrossDexRoute(
        address tokenIn,
        address tokenOut,
        uint256 amountIn,
        address dexAPool,
        address dexBPool
    ) external view returns (uint256 profit, uint256 bestDex) {
        (uint256 amountOutA, ) = _estimateSwap(dexAPool, tokenIn, tokenOut, amountIn);
        (uint256 amountOutB, ) = _estimateSwap(dexBPool, tokenIn, tokenOut, amountIn);

        // This is quote spread (same output token), not absolute PnL.
        if (amountOutA >= amountOutB) {
            profit = amountOutA.sub(amountOutB);
            bestDex = 1;
        } else {
            profit = amountOutB.sub(amountOutA);
            bestDex = 2;
        }
    }

    // Executes multiple Route trades in a single transaction
    function executeBatchRoute(
        address[] calldata tokens,
        uint256[] calldata amounts,
        uint256 minTotalProfit
    ) external onlyAuthorized whenRunning whenNotPaused nonReentrant returns (uint256 totalProfit) {
        uint256 gasStart = gasleft();
        require(tokens.length > 1, "At least 2 tokens required");
        require(tokens.length <= 10, "Too many tokens");
        require(amounts.length > 0, "Missing initial amount");
        require(tokens[0] == WETH, "Base token must be WETH");
        require(tokens[0] == tokens[tokens.length.sub(1)], "Path must close");

        uint256 initialAmount = amounts[0];
        require(initialAmount > 0, "Initial amount must be positive");
        require(initialAmount <= maxTradeSize, "Amount exceeds max trade size");

        uint256 currentAmount = initialAmount;
        for (uint256 i = 0; i < tokens.length.sub(1); i = i.add(1)) {
            address tokenA = tokens[i];
            address tokenB = tokens[i.add(1)];
            require(tokenA != address(0) && tokenB != address(0), "Invalid token");

            address pool = _getPool(tokenA, tokenB, defaultPoolFee);
            require(pool != address(0), "Pool does not exist");

            (uint256 estimatedOut, ) = _estimateSwap(pool, tokenA, tokenB, currentAmount);
            uint256 minOut = estimatedOut.mul(BASIS_POINTS.sub(slippageTolerance)) / BASIS_POINTS;
            currentAmount = _executeSwap(tokenA, tokenB, currentAmount, minOut);
        }

        totalProfit = currentAmount > initialAmount ? currentAmount.sub(initialAmount) : 0;
        require(totalProfit >= minTotalProfit, "Total profit below minimum");
        require(totalProfit >= minProfitThreshold, "Total profit below threshold");

        totalProfitEth = totalProfitEth.add(totalProfit);
        transactionCount = transactionCount.add(1);
        totalVolumeEth = totalVolumeEth.add(initialAmount);
        cumulativeGasUsed = cumulativeGasUsed.add(gasStart.sub(gasleft()));
    }

    // Simulates an Route path without executing
    function simulateAtomicRoute(
        address[] memory path,
        uint256 amountIn
    ) public view returns (uint256 finalAmount, uint256 totalProfit, bool isProfitable) {
        require(path.length >= 3, "Invalid path");
        require(amountIn > 0, "Invalid amount");
        require(path[0] == path[path.length.sub(1)], "Path must return to start token");

        uint256 currentAmount = amountIn;

        for (uint256 i = 0; i < path.length.sub(1); i = i.add(1)) {
            address tokenFrom = path[i];
            address tokenTo = path[i.add(1)];

            if (tokenFrom == address(0) || tokenTo == address(0)) {
                return (0, 0, false);
            }

            address pool = _getPool(tokenFrom, tokenTo, defaultPoolFee);
            if (pool == address(0)) {
                return (0, 0, false);
            }

            (uint256 amountOut, ) = _estimateSwap(pool, tokenFrom, tokenTo, currentAmount);
            if (amountOut == 0) {
                return (0, 0, false);
            }

            currentAmount = amountOut;
        }

        finalAmount = currentAmount;
        totalProfit = finalAmount > amountIn ? finalAmount.sub(amountIn) : 0;
        isProfitable = totalProfit > minProfitThreshold;
    }

    // Calculates market sentiment (bullish/bearish) for multiple tokens
    function calculateMarketSentiment(address[] calldata tokens) external view returns (int256 overallSentiment, uint256 bullishCount, uint256 bearishCount) {
        require(tokens.length > 0, "No tokens provided");

        int256 sentimentSum = 0;
        bullishCount = 0;
        bearishCount = 0;

        for (uint256 i = 0; i < tokens.length; i = i.add(1)) {
            PriceData[] memory history = priceHistory[tokens[i]];
            if (history.length < 2) continue;

            uint256 latestPrice = history[history.length.sub(1)].price;
            uint256 previousPrice = history[0].price;

            int256 tokenSentiment;
            if (latestPrice > previousPrice) {
                uint256 increase = latestPrice.sub(previousPrice).mul(100) / previousPrice;
                tokenSentiment = int256(increase);
                bullishCount = bullishCount.add(1);
            } else if (latestPrice < previousPrice) {
                uint256 decrease = previousPrice.sub(latestPrice).mul(100) / previousPrice;
                tokenSentiment = -int256(decrease);
                bearishCount = bearishCount.add(1);
            } else {
                tokenSentiment = 0;
            }

            sentimentSum = sentimentSum + tokenSentiment;
        }

        if (tokens.length > 0) {
            overallSentiment = sentimentSum / int256(tokens.length);
        } else {
            overallSentiment = 0;
        }
    }

    // Finds the best Route path among multiple intermediate tokens
    function findBestRoutePath(
        address startToken,
        address[] calldata intermediateTokens,
        uint256 amountIn
    ) external view returns (address[] memory bestPath, uint256 maxProfit) {
        require(intermediateTokens.length >= 2, "At least 2 intermediate tokens required");
        require(startToken != address(0), "Invalid start token");
        require(amountIn > 0, "Invalid amount");

        bestPath = new address[](4);
        bestPath[0] = startToken;
        bestPath[3] = startToken;
        maxProfit = 0;

        for (uint256 i = 0; i < intermediateTokens.length.sub(1); i = i.add(1)) {
            address[] memory currentPath = new address[](4);
            currentPath[0] = startToken;
            currentPath[1] = intermediateTokens[i];
            currentPath[2] = intermediateTokens[i.add(1)];
            currentPath[3] = startToken;

            (, uint256 profit, bool profitable) = simulateAtomicRoute(currentPath, amountIn);

            if (profitable && profit > maxProfit) {
                maxProfit = profit;
                bestPath[1] = currentPath[1];
                bestPath[2] = currentPath[2];
            }
        }
    }

    /**
     * @dev Submits Route opportunity found by Python scanner
     * @param tokenIn Input token address
     * @param tokenOut Output token address
     * @param amountIn Amount to trade in wei
     * @param expectedProfit Expected profit in wei
     * @param proof Merkle proof or signature for verification
     * @return opportunityId Unique opportunity identifier
     *
     * Python Opportunity Detector:
     * ```python
     * async def scan_mempool(w3, contract):
     *     pending = await w3.eth.get_pending_transactions()
     *     for tx in pending:
     *         profit = analyze_Route(tx)
     *         if profit > MIN_PROFIT:
     *             submit_opportunity(contract, tx, profit)
     * ```
     */
    function submitPythonOpportunity(
        address tokenIn,
        address tokenOut,
        uint256 amountIn,
        uint256 expectedProfit,
        bytes calldata proof
    ) external onlyAuthorized returns (uint256 opportunityId) {
        require(proof.length > 0, "Proof required");
        bytes32 oppKey = keccak256(abi.encodePacked(
            tokenIn,
            tokenOut,
            amountIn,
            block.number
        ));

        // Prevent duplicate execution from same Python detection
        require(pythonOpportunityExecuted[oppKey] == false, "Opportunity already executed");
        require(pythonOpportunityExpiry[oppKey] < block.timestamp, "Previous opportunity not expired");

        opportunityId = uint256(oppKey);

        // Set expiry for this opportunity (5 minutes)
        pythonOpportunityExpiry[oppKey] = block.timestamp.add(300);
        pythonOpportunityExecuted[oppKey] = true;

        emit RouteOpportunityFound(tokenIn, tokenOut, expectedProfit, block.timestamp);
        emit TradeSignal("OPPORTUNITY_SUBMITTED", tokenIn, expectedProfit, block.timestamp);
    }
